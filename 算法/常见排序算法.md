public class SortTest {
    /**
     * 冒泡
     */
    public static void sort1(int[] arr) {
        int temp;//定义一个临时变量
        for (int i = 0; i < arr.length - 1; i++) {//冒泡趟数
            for (int j = 0; j < arr.length - i - 1; j++) {
                if (arr[j + 1] < arr[j]) {
                    swap(arr, j, j + 1);
                }
            }
        }
    }

    /**
     * 插入
     */
    public static void sort2(int[] arr) {
        int n = arr.length;
        for (int i = 0; i < n; i++) {
            // 寻找元素 arr[i] 合适的插入位置
            for (int j = i; j > 0; j--)
                if (arr[j] < arr[j - 1])
                    swap(arr, j, j - 1);
                else
                    break;
        }
    }

    /**
     * 选择
     */
    public static void sort3(int[] arr) {
        // 总共要经过 N-1 轮比较
        for (int i = 0; i < arr.length - 1; i++) {
            int min = i;
            // 每轮需要比较的次数 N-i
            for (int j = i + 1; j < arr.length; j++) {
                if (arr[j] < arr[min]) {
                    // 记录目前能找到的最小值元素的下标
                    min = j;
                }
            }

            // 将找到的最小值和i位置所在的值进行交换
            if (i != min) {
                swap(arr, i, min);
            }
        }
    }

        int n = arr.length;
        int step = 1;
        while (step < n / 3) { //动态定义间隔序列（步长）
            step = 3 * step + 1;
        }
        while (step >= 1) {
            for (int i = step; i < n; i++) {

                }

    /**
     * 归并
     */
    private static void sort5(int[] array) {
        int[] aux = new int[array.length];
        sort5_1(array, aux, 0, array.length - 1);
    }

    private static void sort5_1(int[] array, int[] aux, int lo, int hi) {
        if (hi <= lo) return;
        int mid = lo + (hi - lo) / 2;
        sort5_1(array, aux, lo, mid);
        sort5_1(array, aux, mid + 1, hi);
        merge(array, aux, lo, mid, hi);
    }

    private static void merge(int[] array, int[] aux, int lo, int mid, int hi) {
        System.arraycopy(array, 0, aux, 0, array.length);
        int i = lo, j = mid + 1;
        for (int k = lo; k <= hi; k++) {
            if (i > mid) array[k] = aux[j++];
            else if (j > hi) array[k] = aux[i++];
            else if (aux[j] < aux[i]) array[k] = aux[j++];
            else array[k] = aux[i++];
        }
    }

    // 交换
    private static void swap(int[] arr, int i, int j) {
        int t = arr[i];
        arr[i] = arr[j];
        arr[j] = t;
    }

    // 获取随机数组
    public static int[] getArr() {
        int[] arr = new int[20000];
        Random r = new Random();
        for (int i = 0; i < arr.length; i++) {
            arr[i] = r.nextInt(10000);
        }
        return arr;
    }
